## Get Test

Retrieve detailed information about a specific test, including its configuration, status, and recent execution history.

<ParamField path="testId" type="string" required>
  The unique identifier or name of the test.
</ParamField>

<ResponseField name="200" body="body">
  <ParamField body="id" path="body.id" type="string">
    Unique identifier for the test.
  </ParamField>
  <ParamField body="name" path="body.name" type="string">
    Name of the test.
  </ParamField>
  <ParamField body="frequency" path="body.frequency" type="string">
    Frequency configuration.
  </ParamField>
  <ParamField body="status" path="body.status" type="string">
    Current status: "active", "paused", "error".
  </ParamField>
  <ParamField body="plan" path="body.plan" type="object">
    The full test plan configuration.
  </ParamField>
  <ParamField body="last_execution" path="body.last_execution" type="object">
    Details of the most recent execution.
    <ParamField body="id" path="body.last_execution.id" type="string">
      Execution identifier.
    </ParamField>
    <ParamField body="status" path="body.last_execution.status" type="string">
      Execution status: "passed", "failed", "error".
    </ParamField>
    <ParamField body="duration" path="body.last_execution.duration" type="integer">
      Execution duration in milliseconds.
    </ParamField>
    <ParamField body="timestamp" path="body.last_execution.timestamp" type="string">
      ISO 8601 timestamp of execution.
    </ParamField>
  </ParamField>
  <ParamField body="next_execution" path="body.next_execution" type="string">
    ISO 8601 timestamp of the next scheduled execution.
  </ParamField>
  <ParamField body="statistics" path="body.statistics" type="object">
    Execution statistics.
    <ParamField body="total_executions" path="body.statistics.total_executions" type="integer">
      Total number of executions.
    </ParamField>
    <ParamField body="success_rate" path="body.statistics.success_rate" type="number">
      Success rate as a decimal (0.0 to 1.0).
    </ParamField>
    <ParamField body="average_duration" path="body.statistics.average_duration" type="integer">
      Average execution duration in milliseconds.
    </ParamField>
  </ParamField>
</ResponseField>

## Example Request

```bash
curl -X GET https://runner.example.com/api/tests/health-check \
  -H "X-API-Key: your-api-key"
```
